void playPianoKey(byte key) {
  String notename[] = {"", "C5", "C5S", "D5", "D5S", "E5", 
    "F5", "F5S", "G5", "G5S", "A5", "A5S", "B5"};
  String type = "piano";
  String filename = type + "/" + notename[key] + ".wav";
  Serial.println(filename);
  if (SD.exists(filename)) Serial.println("found");
  else Serial.println("no file");
  audio.play(filename.c_str());
}

void recordKey(byte key, int dur) {
  if (reclen >= MAX_RECORD_LEN) {
    Serial.println("record limit reached!");
    return;
  }
  rec[reclen].key = key;
  rec[reclen].dur = dur;
  Serial.print("record key: ");
  Serial.print(rec[reclen].key);
  Serial.print(" ");
  Serial.println(rec[reclen].dur);
  reclen++;
}
#define MAX_RECORD_LEN 30
struct Record {
  byte key;
  unsigned int dur;
  Record(): key(0), dur(0) {}
} rec[MAX_RECORD_LEN];
byte reclen = 0;
